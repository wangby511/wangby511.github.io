LRU 实现方式:HashMap + DoubleLinkedList 语言:C++

```javascript
class LRUCache {
struct Node { 
    int key;
    int value; 
    struct Node* next;  
    struct Node* prev;
    Node(int x,int y){
        key = x;
        value = y;
        next = prev = NULL;
    }
};
    map<int,Node*> kv;
    list<Node*> listt;
    Node* head;
    Node* tail;
    int currentsize;
    int cap;
public:
    LRUCache(int capacity) {
        head = new Node(-1,-1);
        tail = new Node(-1,-1);
        head->next = tail;
        head->prev = NULL;
        tail->prev = head;
        tail->next = NULL;
        currentsize = 0;
        cap = capacity;
    }
    
    void MoveToHead(Node* current){
        current->prev->next = current->next;
        current->next->prev = current->prev;
        insertNode(current);
    }
    
    void deleteLastNode(){
        Node* last = tail->prev;
        last->prev->next = tail;
        tail->prev = last->prev;
        last ->next = NULL;
        last ->prev = NULL;
        kv.erase(last->key);
        delete(last);
    }
    
    void insertNode(Node* current){
        Node* first = head->next;
        head->next = current;
        current->next = first;
        first->prev = current;
        current->prev = head;
    }
    
    int get(int key) {
        if(kv.count(key) == 0)return -1;
        Node* current = kv[key];
        MoveToHead(current);
        return current->value;
    }
    
    void print(){
        Node* t = head;
        while(t != NULL){
            cout<<t->value<<" ";
            t = t->next;
        }
        cout<<endl;
    }
    
    void put(int key, int value) {
        if(kv.count(key) == 0){
            Node* newcoming = new Node(key,value);
            insertNode(newcoming);
            kv[key] = newcoming;
            if(++currentsize > cap){
                deleteLastNode();
                currentsize--;
            }
        }else{
            kv[key]->value = value;
            MoveToHead(kv[key]);
        }
    }
};
```
